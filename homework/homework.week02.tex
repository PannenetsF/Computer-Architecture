	add 	$a1, $s0, $zero
	jal 	compare
	sw 	$v0, ($t1)

	# loop end, then restore the stack for caller
	lw	$fp, 48($sp)
	lw	$ra, 44($sp)
	lw	$a0, 40($sp)
	addi 	$sp, $sp, 52
